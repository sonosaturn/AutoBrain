### 1. New Module: `autobrain_core/doctor.py`
This module handles the auto-discovery and environmental patching.

```python
import os
import shutil
import subprocess
import sys
from typing import Optional

class GitNexusDoctor:
    COMMON_WINDOWS_PATHS = [
        r"C:\Program Files\Git\bin\git.exe",
        r"C:\Program Files (x86)\Git\bin\git.exe",
        os.path.expandvars(r"%LocalAppData%\Programs\Git\bin\git.exe"),
        os.path.expandvars(r"%ProgramFiles%\Git\cmd\git.exe")
    ]

    @classmethod
    def validate_git_installation(cls) -> str:
        """
        Validates if git is available. If not, attempts auto-discovery.
        Returns the path to the executable.
        """
        git_path = shutil.which("git")
        
        if git_path:
            return git_path

        # Level 1: Proactive Auto-Discovery
        for potential_path in cls.COMMON_WINDOWS_PATHS:
            if os.path.exists(potential_path):
                # Patch the current process PATH for immediate recovery
                bin_dir = os.path.dirname(potential_path)
                os.environ["PATH"] += os.pathsep + bin_dir
                return potential_path

        # Level 2: Guided Failure
        cls._report_critical_failure()
        sys.exit(2)

    @classmethod
    def _report_critical_failure(cls):
        print("\n" + "="*60)
        print("⚠️  GitNexus Doctor: DEPENDENCY MISSING")
        print("="*60)
        print("The 'git' executable was not found in your system PATH.")
        print("\nWe searched the following common locations:")
        for path in cls.COMMON_WINDOWS_PATHS:
            print(f"  - {path}")
        
        print("\nRESOLUTION:")
        print("1. Install Git from https://git-scm.com/")
        print("2. Ensure 'git' is added to your Environment Variables.")
        print("3. Or run: nexus setup --path \"C:\\path\\to\\git\"")
        print("="*60 + "\n")

def git_execute_wrapper(command_list, **kwargs):
    """
    Wraps subprocess calls to catch WinError 2 and trigger diagnostic.
    """
    try:
        return subprocess.run(command_list, capture_output=True, text=True, check=True, **kwargs)
    except FileNotFoundError as e:
        if e.errno == 2: # WinError 2
            GitNexusDoctor.validate_git_installation()
            # If validation/patching succeeded, retry once
            return subprocess.run(command_list, capture_output=True, text=True, check=True, **kwargs)
        raise e
```
