# Constants for GWCR
SIMILARITY_WEIGHT = 0.7
GRAPH_WEIGHT = 0.3
GHOST_LINK_PENALTY = 0.5  # Multiplier for nodes with 0 frequency

def get_context_for_query(query: str, top_n: int = 10) -> List[dict]:
    """
    Retrieves context using hybrid search (BM25 + Vector) and 
    applies Graph-Weighted Context Reranking (GWCR).
    """
    # 1. Fetch initial candidates (Existing Hybrid Logic)
    # This represents the current existing implementation call
    initial_candidates = _run_hybrid_search_internal(query, limit=top_n * 2)
    
    # 2. Get structural metadata
    try:
        freq_map = get_wikilink_frequencies()
    except Exception as e:
        logging.error(f"GWCR: Failed to fetch wikilink frequencies: {e}")
        freq_map = {}

    reranked_results = []

    # 3. Apply Scoring Heuristic
    for doc in initial_candidates:
        node_id = doc.get('id')
        similarity_score = doc.get('score', 0)
        
        # Get frequency (Centrality)
        link_freq = freq_map.get(node_id, 0)
        
        # Apply Logarithmic scaling to avoid "mega-nodes" overwhelming similarity
        graph_score = math.log(link_freq + 1)
        
        # Calculate Final Score: (Ws * Sim) + (Wg * Graph)
        final_score = (similarity_score * SIMILARITY_WEIGHT) + (graph_score * GRAPH_WEIGHT)
        
        # Ghost Link Filtering / Penalization
        if link_freq == 0:
            final_score *= GHOST_LINK_PENALTY
            
        doc['gwcr_score'] = final_score
        doc['link_frequency'] = link_freq
        reranked_results.append(doc)

    # 4. Sort and return top N
    reranked_results.sort(key=lambda x: x['gwcr_score'], reverse=True)
    
    return reranked_results[:top_n]
