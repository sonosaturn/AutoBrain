def get_context_for_query(query: str, top_n: int = 10, weights: Dict[str, float] = None) -> List[Dict[str, Any]]:
    """
    Retrieves context for a query, reranked by graph centrality (Link Frequency).
    """
    # Default weights
    W_s = weights.get("similarity", 0.7) if weights else 0.7
    W_g = weights.get("graph", 0.3) if weights else 0.3
    
    # 1. Fetch initial candidates from BM25 and Vector Search
    # (Assuming existing internal calls to vector_search and bm25_search)
    candidates = _perform_hybrid_search(query, limit=top_n * 2) 

    # 2. Extract Frequency Map
    # This identifies "Hub" notes based on how often they are linked
    freq_map = get_wikilink_frequencies()

    # 3. Apply Scoring Heuristic and Filter Ghost Links
    reranked_results = []
    for item in candidates:
        node_id = item.get("id")
        similarity_score = item.get("score", 0)
        link_count = freq_map.get(node_id, 0)

        # Logic from remove_isolated_wikilinks: skip if it's a ghost link (0 links)
        if link_count == 0:
            continue

        # Formula: FinalScore = (SimScore * Ws) + (log(Freq + 1) * Wg)
        graph_weight = math.log(link_count + 1)
        final_score = (similarity_score * W_s) + (graph_weight * W_g)
        
        item["final_score"] = final_score
        item["graph_centrality"] = link_count
        reranked_results.append(item)

    # 4. Sort by the new final score and return top_n
    reranked_results.sort(key=lambda x: x["final_score"], reverse=True)
    
    return reranked_results[:top_n]
